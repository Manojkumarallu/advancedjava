package com.mycompany.mavenproject2; import java.sql.Connection; import java.sql.DriverManager; public class Mavenproject2 { public static void main(String[] args) { String url = "jdbc:mysql://localhost:3306/mydb"; String username = "root"; String password = ""; try { // Load MySQL JDBC Driver Class.forName("com.mysql.cj.jdbc.Driver"); // Establish connection Connection con = DriverManager.getConnection(url, username, password); System.out.println("Connected to the database successfully!"); con.close(); } catch (ClassNotFoundException e) { System.err.println("JDBC Driver not found: " + e.getMessage()); } catch (Exception e) { System.err.println("Database error: " + e.getMessage()); } } }

üìù 2. Statement Interface Operations

Objective: Insert and retrieve data using Statement interface

package com.mycompany.mavenproject2; import java.sql.*; public class Mavenproject2 { public static void main(String[] args) { String url = "jdbc:mysql://localhost:3306/mydb"; String username = "root"; String password = ""; try { Class.forName("com.mysql.cj.jdbc.Driver"); Connection con = DriverManager.getConnection(url, username, password); Statement stmt = con.createStatement(); // Insert data String insertQuery = "INSERT INTO students (name, age) VALUES ('Manoj Kumar', 20)"; int rows = stmt.executeUpdate(insertQuery); System.out.println(rows + " record inserted."); // Retrieve data String selectQuery = "SELECT * FROM students"; ResultSet rs = stmt.executeQuery(selectQuery); System.out.println("Student Records:"); while (rs.next()) { System.out.println(rs.getInt("id") + " | " + rs.getString("name") + " | " + rs.getInt("age")); } con.close(); } catch (Exception e) { System.out.println(e); } } }
1 record inserted. Student Records: 1 | Manoj Kumar | 20

üõ†Ô∏è 1. PreparedStatement CRUD Operations

Objective: Perform Insert, Update, Delete using PreparedStatement

package com.mycompany.mavenproject2; import java.sql.*; public class Mavenproject2 { public static void main(String[] args) { String url = "jdbc:mysql://localhost:3306/mydb"; String user = "root"; String password = ""; try { Class.forName("com.mysql.cj.jdbc.Driver"); Connection con = DriverManager.getConnection(url, user, password); System.out.println("Connected!"); // INSERT String insertSQL = "INSERT INTO Person (PersonID, FirstName, City) VALUES (?, ?, ?)"; PreparedStatement insertStmt = con.prepareStatement(insertSQL); insertStmt.setInt(1, 101); insertStmt.setString(2, "Manoj Kumar"); insertStmt.setString(3, "Hyderabad"); int insertCount = insertStmt.executeUpdate(); System.out.println(insertCount + " row inserted."); // UPDATE String updateSQL = "UPDATE Person SET City=? WHERE PersonID=?"; PreparedStatement updateStmt = con.prepareStatement(updateSQL); updateStmt.setString(1, "Bangalore"); updateStmt.setInt(2, 101); int updateCount = updateStmt.executeUpdate(); System.out.println(updateCount + " row updated."); // DELETE String deleteSQL = "DELETE FROM Person WHERE PersonID=?"; PreparedStatement deleteStmt = con.prepareStatement(deleteSQL); deleteStmt.setInt(1, 101); int deleteCount = deleteStmt.executeUpdate(); System.out.println(deleteCount + " row deleted."); con.close(); } catch (Exception e) { System.out.println("Error : " + e.getMessage()); } } }
Connected! 1 row inserted. 1 row updated. 1 row deleted.

üß≠ 2. ResultSet Navigation Methods

Objective: Explore ResultSet navigation and tabular data formatting

package com.mycompany.mavenproject2; import java.sql.*; public class Mavenproject2 { public static void main(String[] args) { String url = "jdbc:mysql://localhost:3306/mydb"; String user = "root"; String password = ""; try { Class.forName("com.mysql.cj.jdbc.Driver"); Connection con = DriverManager.getConnection(url, user, password); Statement stmt = con.createStatement( ResultSet.TYPE_SCROLL_INSENSITIVE, ResultSet.CONCUR_READ_ONLY ); ResultSet rs = stmt.executeQuery("SELECT PersonID, FirstName, City FROM Person"); // Print table header System.out.println("--------------------------------------------------"); System.out.printf("%-10s %-15s %-15s\n", "ID", "Name", "City"); System.out.println("--------------------------------------------------"); // Move to first record if (rs.first()) { do { System.out.printf("%-10d %-15s %-15s\n", rs.getInt("PersonID"), rs.getString("FirstName"), rs.getString("City") ); } while (rs.next()); } System.out.println("--------------------------------------------------"); // Navigation examples if (rs.last()) { System.out.println("Last Record ‚Üí " + rs.getString("FirstName")); } if (rs.first()) { System.out.println("First Record ‚Üí " + rs.getString("FirstName")); } if (rs.absolute(2)) { System.out.println("2nd Row ‚Üí " + rs.getString("FirstName")); } con.close(); } catch (Exception e) { System.out.println("Error : " + e); } } }
-------------------------------------------------- ID Name City -------------------------------------------------- 101 Manoj Kumar Hyderabad 102 Rahul Bangalore 103 Priya Mumbai -------------------------------------------------- Last Record ‚Üí Priya First Record ‚Üí Manoj Kumar 2nd Row ‚Üí Rahul

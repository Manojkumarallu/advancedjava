package com.mycompany.mavenproject2;
import java.sql.Connection;
import java.sql.DriverManager;

public class Mavenproject2 {
    public static void main(String[] args) {
        // Step 1: Define connection parameters
        String url = "jdbc:mysql://localhost:3306/mydb";
        String username = "root";
        String password = "";
        
        try {
            // Step 2: Load MySQL JDBC Driver
            Class.forName("com.mysql.cj.jdbc.Driver");

            Connection con = DriverManager.getConnection(url, username, password);

            con.close();

            System.err.println("JDBC Driver not found: " + e.getMessage());
        } catch (Exception e) {
            System.err.println("Database error: " + e.getMessage());
        }
    }
}

Mavenproject2.java - Statement Operations 
package com.mycompany.mavenproject2;
import java.sql.*;

public class Mavenproject2 {
    public static void main(String[] args) {
        // Connection parameters
        String url = "jdbc:mysql://localhost:3306/mydb";
        String username = "root";
        String password = "";
        
        try {
            // Step 1: Load driver and establish connection
            Class.forName("com.mysql.cj.jdbc.Driver");
            Connection con = DriverManager.getConnection(url, username, password);
            
            // Step 2: Create Statement object
            Statement stmt = con.createStatement();
            
            // Step 3: Execute INSERT operation
            String insertQuery = "INSERT INTO students (name, age) VALUES ('Manoj Kumar', 20)";
            int rows = stmt.executeUpdate(insertQuery);
            System.out.println(rows + " record inserted.");
            
            // Step 4: Execute SELECT operation
            String selectQuery = "SELECT * FROM students";
            ResultSet rs = stmt.executeQuery(selectQuery);
            
            // Step 5: Process ResultSet
            System.out.println("Student Records:");
            while (rs.next()) {
                System.out.println(rs.getInt("id") + " | " +
                                 rs.getString("name") + " | " +
                                 rs.getInt("age"));
            }
            
            // Step 6: Close resources
            rs.close();
            stmt.close();
            con.close();
        } catch (Exception e) {
            System.out.println("Error: " + e.getMessage());
        }
    }
}

Mavenproject2.java - PreparedStatement CRUD 
package com.mycompany.mavenproject2;
import java.sql.*;

public class Mavenproject2 {
    public static void main(String[] args) {
        // Database connection parameters
        String url = "jdbc:mysql://localhost:3306/mydb";
        String user = "root";
        String password = "";
        
        try {
            // Step 1: Load JDBC driver and establish connection
            Class.forName("com.mysql.cj.jdbc.Driver");
            Connection con = DriverManager.getConnection(url, user, password);
            System.out.println("Connected!");
            
            // Step 2: INSERT Operation with PreparedStatement
            String insertSQL = "INSERT INTO Person (PersonID, FirstName, City) VALUES (?, ?, ?)";
            PreparedStatement insertStmt = con.prepareStatement(insertSQL);
            insertStmt.setInt(1, 101);              // Set PersonID parameter
            insertStmt.setString(2, "Rahul");        // Set FirstName parameter
            insertStmt.setString(3, "Hyderabad");    // Set City parameter
            int insertCount = insertStmt.executeUpdate();
            System.out.println(insertCount + " row inserted.");
            
            // Step 3: UPDATE Operation with PreparedStatement
            String updateSQL = "UPDATE Person SET City=? WHERE PersonID=?";
            PreparedStatement updateStmt = con.prepareStatement(updateSQL);
            updateStmt.setString(1, "Bangalore");    // New city value
            updateStmt.setInt(2, 101);               // WHERE condition
            int updateCount = updateStmt.executeUpdate();
            System.out.println(updateCount + " row updated.");
            
            // Step 4: DELETE Operation with PreparedStatement
            String deleteSQL = "DELETE FROM Person WHERE PersonID=?";
            PreparedStatement deleteStmt = con.prepareStatement(deleteSQL);
            deleteStmt.setInt(1, 101);               // WHERE condition
            int deleteCount = deleteStmt.executeUpdate();
            System.out.println(deleteCount + " row deleted.");
            
            // Step 5: Close connection and free resources
            con.close();
        } catch (Exception e) {
            System.out.println("Error : " + e.getMessage());
        }
    }
}

  • Type Safety: setInt(), setString() ensure correct data types

Performance Advantages

  • Optimization: Database can optimize execution plan
Parameter Binding

  • Index-based: Parameters numbered from 1
  • Type-specific: setInt(), setString(), setDate() methods

ResultSet Navigation & Tabular Display

Advanced cursor movement and data formatting techniques

ResultSet Navigation Methods

rs.first()

Moves cursor to the first row of the ResultSet

rs.last()

Moves cursor to the last row of the ResultSet

rs.absolute(n)

Moves cursor to the specified row number

rs.next()

Moves cursor to the next row in sequence

Mavenproject2.java - ResultSet Navigation 
package com.mycompany.mavenproject2;
import java.sql.*;

public class Mavenproject2 {
    public static void main(String[] args) {
        // Database connection parameters
        String url = "jdbc:mysql://localhost:3306/mydb";
        String user = "root";
        String password = "";
        
        try {
            // Step 1: Establish database connection
            Class.forName("com.mysql.cj.jdbc.Driver");
            Connection con = DriverManager.getConnection(url, user, password);
            
            // Step 2: Create scrollable ResultSet
            Statement stmt = con.createStatement(
                ResultSet.TYPE_SCROLL_INSENSITIVE,  // Allows backward/forward movement
                ResultSet.CONCUR_READ_ONLY          // Read-only access
            );
            
            // Step 3: Execute query and get ResultSet
            ResultSet rs = stmt.executeQuery("SELECT PersonID, FirstName, City FROM Person");
            
            // Step 4: Display data in tabular format
            System.out.println("--------------------------------------------------");
            System.out.printf("%-10s %-15s %-15s\n", "ID", "Name", "City");
            System.out.println("--------------------------------------------------");
            
            // Step 5: Navigate to first record and display all
            if (rs.first()) {
                do {
                    System.out.printf("%-10d %-15s %-15s\n",
                        rs.getInt("PersonID"),      // Get integer value
                        rs.getString("FirstName"),  // Get string value
                        rs.getString("City")        // Get string value
                    );
                } while (rs.next());  // Move to next record
            }
            System.out.println("--------------------------------------------------");
            
            // Step 6: Demonstrate navigation methods
            if (rs.last()) {
                System.out.println("Last Record → " + rs.getString("FirstName"));
            }
            
            if (rs.first()) {
                System.out.println("First Record → " + rs.getString("FirstName"));
            }
            
            if (rs.absolute(2)) {
                System.out.println("2nd Row → " + rs.getString("FirstName"));
            }
            
            // Step 7: Close resources
            rs.close();
            stmt.close();
            con.close();
        } catch (Exception e) {
            System.out.println("Error : " + e.getMessage());
        }
    }
}

DELIMITER //
CREATE PROCEDURE GetStudentById(IN student_id INT)
BEGIN
    SELECT * FROM students WHERE id = student_id;
END //
DELIMITER ;

-- Procedure 2: IN and OUT parameters
DELIMITER //
CREATE PROCEDURE GetStudentCount(IN min_age INT, OUT total_count INT)
BEGIN
    SELECT COUNT(*) INTO total_count 
    FROM students 
    WHERE age >= min_age;
END //
DELIMITER ;

-- Procedure 3: Complex procedure with multiple operations
DELIMITER //
CREATE PROCEDURE ManageStudent(
    IN action_type VARCHAR(10),
    IN student_name VARCHAR(50),
    IN student_age INT,
    IN student_city VARCHAR(50),
    OUT result_message VARCHAR(100)
)
BEGIN
    DECLARE student_exists INT DEFAULT 0;
    
    IF action_type = 'INSERT' THEN
        INSERT INTO students (name, age, city) 
        VALUES (student_name, student_age, student_city);
        SET result_message = 'Student added successfully';
        
    ELSEIF action_type = 'CHECK' THEN
        SELECT COUNT(*) INTO student_exists 
        FROM students 
        WHERE name = student_name;
        
        IF student_exists > 0 THEN
            SET result_message = 'Student exists in database';
        ELSE
            SET result_message = 'Student not found';
        END IF;
    END IF;
END //
DELIMITER ;

02

CallableStatement Implementation

Java code to execute stored procedures with parameters

Mavenproject2.java - CallableStatement 
package com.mycompany.mavenproject2;
import java.sql.*;

public class Mavenproject2 {
    public static void main(String[] args) {
        // Database connection parameters
        String url = "jdbc:mysql://localhost:3306/mydb";
        String user = "root";
        String password = "";
        
        try {
            // Step 1: Establish database connection
            Class.forName("com.mysql.cj.jdbc.Driver");
            Connection con = DriverManager.getConnection(url, user, password);
            System.out.println("Connected to database!");
            
            // Step 2: Call procedure with IN parameter
            System.out.println("\n=== Calling GetStudentById Procedure ===");
            CallableStatement cs1 = con.prepareCall("{call GetStudentById(?)}");
            cs1.setInt(1, 1); // Set student ID parameter
            
            ResultSet rs = cs1.executeQuery();
            System.out.println("Student Details:");
            while (rs.next()) {
                System.out.println("ID: " + rs.getInt("id"));
                System.out.println("Name: " + rs.getString("name"));
                System.out.println("Age: " + rs.getInt("age"));
                System.out.println("City: " + rs.getString("city"));
            }
            
            // Step 3: Call procedure with IN and OUT parameters
            System.out.println("\n=== Calling GetStudentCount Procedure ===");
            CallableStatement cs2 = con.prepareCall("{call GetStudentCount(?, ?)}");
            cs2.setInt(1, 18);                          // IN parameter: minimum age
            cs2.registerOutParameter(2, Types.INTEGER); // OUT parameter: count
            
            cs2.execute();
            int studentCount = cs2.getInt(2); // Get OUT parameter value
            System.out.println("Students aged 18 or above: " + studentCount);
            
            // Step 4: Call complex procedure with multiple parameters
            System.out.println("\n=== Calling ManageStudent Procedure ===");
            CallableStatement cs3 = con.prepareCall("{call ManageStudent(?, ?, ?, ?, ?)}");
            
            // Insert operation
            cs3.setString(1, "INSERT");           // action_type
            cs3.setString(2, "Priya Sharma");     // student_name
            cs3.setInt(3, 22);                    // student_age
            cs3.setString(4, "Mumbai");           // student_city
            cs3.registerOutParameter(5, Types.VARCHAR); // result_message
            
            cs3.execute();
            String insertResult = cs3.getString(5);
            System.out.println("Insert Result: " + insertResult);
            
            // Check operation
            cs3.setString(1, "CHECK");
            cs3.setString(2, "Priya Sharma");
            cs3.setString(3, null);
            cs3.setString(4, null);
            cs3.registerOutParameter(5, Types.VARCHAR);
            
            cs3.execute();
            String checkResult = cs3.getString(5);
            System.out.println("Check Result: " + checkResult);
            
            // Step 5: Close resources
            rs.close();
            cs1.close();
            cs2.close();
            cs3.close();
            con.close();
            
        } catch (Exception e) {
            System.out.println("Error: " + e.getMessage());
            e.printStackTrace();
        }
    }
}
Parameter Types Explained
Parameter Type Description Java Method Example
IN Input values passed to procedure setXXX() cs.setInt(1, 101)
OUT Output values returned by procedure registerOutParameter() + getXXX() cs.registerOutParameter(2, Types.INTEGER)
INOUT Both input and output parameter setXXX() + registerOutParameter() cs.setInt(1, 101); cs.registerOutParameter(1, Types.INTEGER)

Master RequestDispatcher, sendRedirect & Attribute Passing

RequestDispatcher

Core Concepts Overview

RequestDispatcher

Server-side forwarding mechanism that transfers control to another servlet while maintaining request attributes and parameters.

sendRedirect

Client-side redirection that sends HTTP redirect response to browser, creating a new request to the target URL.

Attribute Passing

Mechanism to share data between servlets using request attributes with setAttribute() and getAttribute() methods.

Forward vs Redirect Comparison

Aspect RequestDispatcher (Forward) sendRedirect
Location Server-side Client-side
URL Change No change in browser URL URL changes in browser
Request Object Same request object New request object
Attributes Attributes preserved Attributes lost
Performance Faster (single request) Slower (two requests)

RequestDispatcher & sendRedirect Demo

FirstServlet.java 
package com.mycompany.servlets;
import java.io.IOException;
import javax.servlet.*;
import javax.servlet.http.*;

public class FirstServlet extends HttpServlet {
    @Override
    protected void doGet(HttpServletRequest req, HttpServletResponse res)
            throws ServletException, IOException {
        
        // Set attribute to pass data to forwarded servlet
        req.setAttribute("message", "Forwarded using RequestDispatcher!");
        req.setAttribute("rollNumber", "238W1A1269");
        req.setAttribute("developer", "Manoj Kumar");
        
        // Get RequestDispatcher for forwarding
        RequestDispatcher rd = req.getRequestDispatcher("forward");
        
        // Forward request to ForwardServlet
        rd.forward(req, res);
    }
}

public class AttributeSender extends HttpServlet {
